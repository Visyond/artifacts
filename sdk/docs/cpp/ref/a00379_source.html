291 inline bool approxEqual(double a, double b, sal_Int16 nPrec)
292 {
293  if ( a == b )
294  return true;
295  double x = a - b;
296  return (x < 0.0 ? -x : x)
297  < ((a < 0.0 ? -a : a) * (1.0 / (pow(2.0, nPrec))));
298 }
